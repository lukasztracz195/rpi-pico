# 02 - loop, delay


```python
from machine import Pin
led = Pin(25, Pin.OUT)

import time

while True:
    led.value(1)
    time.sleep_ms(500)
    led.value(0)
    time.sleep_ms(500)
```

from machine import Pin
led = Pin(25, Pin.OUT)


def ledMyBlink():
    led.value(1)
    time.sleep_ms(100)
    led.value(0)
    time.sleep_ms(100)
    led.value(1)
    time.sleep_ms(450)
    led.value(0)
    time.sleep_ms(350)
    

import time

while True:
    ledMyBlink()
